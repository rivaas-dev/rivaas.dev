Functional options everywhere. Configure observability, OpenAPI, health checks, and more‚Äîall the same way.

main.go 
a := app.New(
    // Service identity
    app.WithServiceName("my-api"),
    app.WithServiceVersion("v1.2.3"),
    
    // Observability: metrics + tracing + logging
    app.WithObservability(
        app.WithMetrics(),
        app.WithTracing(tracing.WithOTLP("localhost:4317")),
        app.WithLogging(logging.WithJSONHandler()),
    ),
    
    // OpenAPI documentation
    app.WithOpenAPI(openapi.WithTitle("My API")),
    
    // Health & readiness checks
    app.WithHealthEndpoints(
        app.WithReadinessCheck("database", dbPingCheck),
    ),
)
